    <script>
        // --- 1. JARVIS VOICE SYSTEM ---
        let isJarvisActive = false;
        const synth = window.speechSynthesis;
        let voices = [];

        // Load Voices
        window.speechSynthesis.onvoiceschanged = () => {
            voices = synth.getVoices();
        };

        function getJarvisVoice() {
            // Try to find a robotic/authoritative male voice
            return voices.find(v => v.name.includes("Google UK English Male") || v.name.includes("Microsoft David")) || voices[0];
        }

        function speak(text) {
            if (!isJarvisActive) return;

            // Visualizer On
            document.getElementById('voice-viz').classList.remove('opacity-0');
            
            const utterance = new SpeechSynthesisUtterance(text);
            utterance.voice = getJarvisVoice();
            utterance.rate = 1.0; 
            utterance.pitch = 0.9; // Slightly deeper for Jarvis effect
            
            utterance.onend = () => {
                // Visualizer Off
                document.getElementById('voice-viz').classList.add('opacity-0');
            };

            synth.speak(utterance);
        }

        // --- 2. SOUND EFFECTS (SFX) ---
        const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        
        function playTouchSound() {
            if (!isJarvisActive) return;
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            
            osc.type = 'sine';
            osc.frequency.setValueAtTime(800, audioCtx.currentTime);
            osc.frequency.exponentialRampToValueAtTime(300, audioCtx.currentTime + 0.1);
            
            gain.gain.setValueAtTime(0.1, audioCtx.currentTime);
            gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 0.1);
            
            osc.start();
            osc.stop(audioCtx.currentTime + 0.1);
        }

        function playAlertSound() {
            if (!isJarvisActive) return;
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            
            osc.type = 'square';
            osc.frequency.setValueAtTime(440, audioCtx.currentTime);
            osc.frequency.setValueAtTime(880, audioCtx.currentTime + 0.2);
            
            gain.gain.setValueAtTime(0.1, audioCtx.currentTime);
            gain.gain.linearRampToValueAtTime(0, audioCtx.currentTime + 0.5);
            
            osc.start();
            osc.stop(audioCtx.currentTime + 0.5);
        }

        // --- 3. GEMINI AI INTEGRATION ---
        async function askJarvis() {
            const apiKey = document.getElementById('gemini-key').value.trim();
            const query = document.getElementById('user-query').value.trim();
            
            if (!apiKey) {
                alert("ACCESS DENIED: Please enter Gemini API Key.");
                speak("Access Denied. Please provide security credentials.");
                return;
            }
            if (!query) return;

            log(`USER: ${query}`, 'info');
            document.getElementById('user-query').value = '';
            
            // Show thinking state
            speak("Processing request.");
            
            const prompt = `
                You are JARVIS, an advanced AI trading assistant for Boss Manikandan. 
                Keep responses short, robotic, and authoritative.
                Current Market Data: Nifty @ ${currentPrice.toFixed(2)}, Velocity: ${document.getElementById('val-velocity').innerText}.
                User asks: "${query}"
            `;

            try {
                const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
                const response = await fetch(url, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] })
                });
                
                const data = await response.json();
                if(data.candidates && data.candidates[0].content) {
                    const reply = data.candidates[0].content.parts[0].text;
                    log(`JARVIS: ${reply}`, 'success');
                    speak(reply);
                }
            } catch (e) {
                log("CONNECTION ERROR", 'danger');
                speak("Connection lost. Unable to reach servers.");
            }
        }

        // --- 4. SYSTEM CONTROLS ---
        function toggleJarvis() {
            isJarvisActive = !isJarvisActive;
            const status = document.getElementById('voice-status');
            
            if(isJarvisActive) {
                status.innerText = "JARVIS ONLINE";
                status.classList.add("text-emerald-400");
                status.classList.remove("text-cyan-400");
                speak("Systems Online. Welcome back, Boss Manikandan. All protocols secured.");
                log("Voice Module Activated.", "success");
            } else {
                status.innerText = "SYSTEM MUTED";
                status.classList.remove("text-emerald-400");
                status.classList.add("text-cyan-400");
                synth.cancel();
            }
        }

        function log(msg, type) {
            const term = document.getElementById('terminal');
            const div = document.createElement('div');
            const color = type === 'success' ? 'text-emerald-400' : (type === 'warning' ? 'text-amber-400' : 'text-cyan-300');
            div.innerHTML = `<span class="text-slate-500">[LOG]</span> <span class="${color}">> ${msg}</span>`;
            term.prepend(div);
        }

        // --- 5. SIMULATION LOOP ---
        let currentPrice = 19500;
        let vwap = 19500;
        const maxPoints = 50;
        const labels = Array(maxPoints).fill('');
        const priceData = Array(maxPoints).fill(19500);
        const vwapData = Array(maxPoints).fill(19500);

        // Chart Init
        const ctx = document.getElementById('mainChart').getContext('2d');
        const chart = new Chart(ctx, {
            type: 'line',
            data: {
                labels: labels,
                datasets: [
                    {
                        label: 'Price',
                        data: priceData,
                        borderColor: '#00f3ff',
                        backgroundColor: 'rgba(0, 243, 255, 0.1)',
                        borderWidth: 2,
                        tension: 0.4,
                        pointRadius: 0,
                        fill: true
                    },
                    {
                        label: 'VWAP',
                        data: vwapData,
                        borderColor: '#ef4444',
                        borderWidth: 1,
                        borderDash: [5, 5],
                        pointRadius: 0
                    }
                ]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                animation: false,
                plugins: { legend: { display: false } },
                scales: {
                    x: { display: false },
                    y: { grid: { color: '#1e293b' }, ticks: { color: '#64748b' } }
                }
            }
        });

        function updateSimulation() {
            // Math
            const move = (Math.random() - 0.5) * 10;
            currentPrice += move;
            vwap = (vwap * 19 + currentPrice) / 20;
            const velocity = move;
            
            // UI Update
            document.getElementById('nifty-price').innerText = currentPrice.toFixed(2);
            document.getElementById('val-velocity').innerText = velocity.toFixed(2);
            
            // Logic Trigger
            if (velocity > 4) {
                triggerSignal("BUY CE", "High Velocity Detected");
            } else if (velocity < -4) {
                triggerSignal("BUY PE", "Momentum Crash Detected");
            } else {
                document.getElementById('ai-decision').innerText = "SCANNING...";
                document.getElementById('ai-decision').className = "text-3xl font-black text-slate-500 animate-pulse";
            }

            // Chart Update
            priceData.push(currentPrice); priceData.shift();
            vwapData.push(vwap); vwapData.shift();
            chart.update();
        }

        function triggerSignal(type, reason) {
            const el = document.getElementById('ai-decision');
            const color = type.includes("CE") ? "text-emerald-400" : "text-red-500";
            
            if (el.innerText !== type) { // Only if changed
                el.innerText = type;
                el.className = `text-4xl font-black ${color} animate-bounce`;
                
                // JARVIS SPEAK
                playAlertSound();
                if(Math.random() > 0.7) { // Don't speak too often
                    speak(`Boss! ${reason}. Suggesting ${type}.`);
                    log(`SIGNAL: ${type} | ${reason}`, "warning");
                }
            }
        }

        // Start Loop
        setInterval(updateSimulation, 1000);

    </script>
